public class DrawRectangleWithStar 
{
    
    public static void main(String args[])
    {
        for (int row = 1; row <= 3; row++)
        {
            for (int column = 1; column <= 6; column++)
            {
                System.out.print("*");
            }
            System.out.println();
        }
    }
    
}
